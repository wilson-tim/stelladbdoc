AS

/* A type to represent a SOAP RPC request */
TYPE request 
IS RECORD 
(method     VARCHAR2(256),     
 namespace  VARCHAR2(256),     
 body       CLOB);    
 --body       VARCHAR2(32767));    
 
 /* A type to represent a SOAP RPC response */   
 TYPE response 
 IS RECORD 
 (doc xmltype);    
 
 /*    * Create a new SOAP RPC request.    */   
 FUNCTION new_request(method    IN VARCHAR2,
                      namespace IN VARCHAR2)
          RETURN request;    
 
 /*    * Add a simple parameter to the SOAP RPC request.    */   
 PROCEDURE add_parameter(req   IN OUT NOCOPY request,
                         name  IN VARCHAR2,
                         type  IN VARCHAR2,                           
                         value IN VARCHAR2);    

 /*    * Make the SOAP RPC call.    */   
 FUNCTION invoke(req          IN OUT NOCOPY request,
                 url          IN VARCHAR2,
                 action       IN VARCHAR2,
                 username     IN VARCHAR2 DEFAULT NULL,
                 password     IN VARCHAR2 DEFAULT NULL,                   
                 http_version IN VARCHAR2 DEFAULT utl_http.HTTP_VERSION_1_1) 
        RETURN response;    

 FUNCTION invoke_varchar2(req   IN OUT NOCOPY request,
                   url          IN VARCHAR2,
                   action       IN VARCHAR2,
                   username     IN VARCHAR2 DEFAULT NULL,
                   password     IN VARCHAR2 DEFAULT NULL,                   
                   http_version IN VARCHAR2 DEFAULT utl_http.HTTP_VERSION_1_1) 
       RETURN VARCHAR2;

 FUNCTION invoke_soap_message(req   IN OUT NOCOPY request,
                   url              IN VARCHAR2,
                   action           IN VARCHAR2,
                   soap_message     IN VARCHAR2,
                   username         IN VARCHAR2 DEFAULT NULL,
                   password         IN VARCHAR2 DEFAULT NULL,                   
                   http_version     IN VARCHAR2 DEFAULT utl_http.HTTP_VERSION_1_1)  
       RETURN response; 

 /*    * Retrieve the simple return value of the SOAP RPC call.    */   
 FUNCTION get_return_value(resp      IN OUT NOCOPY response,
                           name      IN VARCHAR2,                             
                           namespace IN VARCHAR2) 
       RETURN VARCHAR2;  

 /* Check Soap response for a fault node   */
 FUNCTION check_fault(resp           IN OUT NOCOPY response)
       RETURN NUMBER;

END p_soap_utils;
