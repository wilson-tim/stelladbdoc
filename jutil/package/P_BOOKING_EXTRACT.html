PACKAGE p_Booking_Extract IS

  -- Author  : JDURNFORD
  -- Created : 12/12/2006 11:45:35
  -- Purpose : Extract booking data from the Warehouse relating to either 
  --           Mainstream or Travel-Link

  -- Public type declarations
  TYPE return_refcursor is REF CURSOR;

  v_Start_Errm          CONSTANT  VARCHAR2(40) := '%%FCE_START_ERRM%%';
  v_End_Errm            CONSTANT  VARCHAR2(40) := '%%FCE_END_ERRM%%';
  v_Start_ErrCode       CONSTANT  VARCHAR2(40) := '%%FCE_START_ERR_CODE%%';
  v_End_ErrCode         CONSTANT  VARCHAR2(40) := '%%FCE_END_ERR_CODE%%';
  
  -- Main Function to return a cursor containing the accommodation details
  FUNCTION Extract_Accom_Details (p_CompanyCode   IN VARCHAR2,
                                  p_BookingRef    IN VARCHAR2,
                                  p_SeasonType    IN VARCHAR2,
                                  p_SeasonYear    IN VARCHAR2)
  RETURN return_refcursor;


  -- Main Function to return a cursor containing the booking & flight details
  FUNCTION Extract_Flight_Details (p_CompanyCode   IN VARCHAR2,
                                   p_BookingRef    IN VARCHAR2,
                                   p_SeasonType    IN VARCHAR2,
                                   p_SeasonYear    IN VARCHAR2)
  RETURN return_refcursor;
  

  -- Main Procedure to return both flight & accommodation cursors
  PROCEDURE Extract_Details (p_CompanyCode   IN VARCHAR2,
                             p_BookingRef    IN VARCHAR2,
                             p_SeasonType    IN VARCHAR2,
                             p_SeasonYear    IN VARCHAR2,
                             p_BookingCur    IN OUT return_refcursor,
                             p_FlightCur     IN OUT return_refcursor,
                             p_AccomCur      IN OUT return_refcursor);
  
  
END p_Booking_Extract;
