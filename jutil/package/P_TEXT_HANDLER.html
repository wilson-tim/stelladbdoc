PACKAGE p_text_handler IS

  -- Author  : LASHTON
  -- Created : 21/05/2004 11:44:10
  -- Purpose : various text handling utilities

  g_version      NUMBER := 1.00;
  g_package_name VARCHAR2(40) := 'p_text_handler';
  g_statement    NUMBER := 0;
  g_sqlerrm      VARCHAR2(500);
  g_sqlcode      CHAR(20);

  ------------------------------------------------------------------------------------ 
  ------------------------------------------------------------------------------------ 
  -- NAME:       wwrap
  -- PURPOSE:    done many times before, this is yet-another wordwrapper but
  --         much smaller than any ive seen
  --
  --         handles LEFT justify, RIGHT justify, CENTER
  --
  -- NOTES:     PARAMETERS:
  --             #param  p_textin = text to format; up to clob size
  --             #param  p_rmargin = wrap at what col position -- i suggest you set this high in case a line is longer than this length without any breaks in it
  --             #param  p_pad = pad the text after wrapping with what default to SPACE
  --             #param  p_worddelim = words delimeted by what? defaults to SPACE
  --             #param  p_just = justify how? Left,Right,Center  
  FUNCTION wordwrap(p_textin    IN CLOB DEFAULT NULL,
                    p_rmargin   IN INTEGER DEFAULT 120,
                    p_pad       IN VARCHAR2 DEFAULT ' ',
                    p_worddelim IN VARCHAR2 DEFAULT ' ',
                    p_just      IN VARCHAR2 DEFAULT 'LEFT') RETURN CLOB;

  /*   create a delimited list of values from a piece of sql 
  overloaded function to create a ref cursor which can then be passed to concat_list function that accepts 
     cursors */
  -- #param p_cur_sql  The piece of sql to be fired and the column to be returned in a list - mandatory
  -- #param p_concat_char  The character to be used as a delimiter e.g. a comma. Leave blank/null for no delimiter - optional   
  FUNCTION concat_list(p_cur_sql     VARCHAR2,
                       p_concat_char IN CHAR DEFAULT '') RETURN VARCHAR2;

  /*   create a delimited list of values from a piece of sql  passed as a ref cursor
  there is also an overloaded version of this which accepts a sql statement*/
  -- #param p_cur_sql  The piece of sql to be fired and the column to be returned in a list - mandatory
  -- #param p_concat_char  The character to be used as a delimiter e.g. a comma. Leave blank/null for no delimiter - optional   
  FUNCTION concat_list(p_cur         sys_refcursor,
                       p_concat_char IN CHAR DEFAULT '') RETURN CLOB;

END p_text_handler;
