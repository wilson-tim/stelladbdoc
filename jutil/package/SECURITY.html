PACKAGE security AS

  -- Author  : leigh ashton
  -- Created : 03/03/03
  -- Purpose : Defines the procedures and functions for the security system
  --         : used by on-line applications e.g. Stella.  
  --         

  -- Public type declarations
  TYPE return_refcursor IS REF CURSOR;

  g_statement       NUMBER := 0;
  g_sqlerrm         VARCHAR2(500);
  g_sqlcode         CHAR(20);
  g_error_statement VARCHAR2(150) := '';

  ----------------------------------------------------------
  ----------------------------------------------------------
  /* return resultset of a single users security details  */
  FUNCTION get_single_user_details(p_user_name CHAR) RETURN return_refcursor;
  ----------------------------------------------------------
  ----------------------------------------------------------

  /* return resultset of all users security details 
  or restrict it to one application if that parameter passed */
  /* p_application_key is not mandatory */
  /* return all users that are assigned a role that belongs to that application */
  FUNCTION get_all_user_details(p_application_key IN CHAR)
    RETURN return_refcursor;

  ----------------------------------------------------------
  ----------------------------------------------------------
  /*update or insert user security details    */
  FUNCTION update_user_details(p_user_name    CHAR,
                               p_initials     CHAR,
                               p_password     CHAR,
                               p_disabled_ind CHAR) RETURN CHAR;

  ----------------------------------------------------------
  ----------------------------------------------------------    
  /* return an encrypted version of a passed string , used for hiding password from
  people selecting security details from sql */
  FUNCTION encrypt_password(p_input_string IN CHAR) RETURN VARCHAR2;

  ----------------------------------------------------------
  ----------------------------------------------------------
  /* return an decrypted version of a passed string , used for hiding password from
  people selecting security details from sql */

  FUNCTION decrypt_password(p_input_string IN CHAR) RETURN VARCHAR2;

  ----------------------------------------------------------
  ----------------------------------------------------------
  /* update the password of a particular user, 
  password is automatically encrypted */
  FUNCTION set_password(p_user_name    IN CHAR,
                        p_new_password IN CHAR,
                        p_old_password IN CHAR) RETURN VARCHAR2;

  ----------------------------------------------------------
  ----------------------------------------------------------
  /* delete a single user PLUS all allocated roles */
  FUNCTION delete_user(p_user_name IN CHAR) RETURN VARCHAR2;

  ----------------------------------------------------------
  ----------------------------------------------------------
  -- returns cursor/result set of all roles that are allocated to a user
  -- p_user_name is mandatory, p_application_key can be null
  -- if p_application_key is supplied, then only roles associated with that user 
  -- within a particular application are returned
  --if p_application_key is null, then every role in all applications that have been assigned to this user are returned
  FUNCTION get_user_security_roles(p_user_name       IN CHAR,
                                   p_application_key IN CHAR)
    RETURN security.return_refcursor;

  ----------------------------------------------------------
  ----------------------------------------------------------
  -- return cursor/result set of all roles that are available to be allocated to a user
  FUNCTION get_app_security_roles(p_application_key IN CHAR)
    RETURN security.return_refcursor;

  ----------------------------------------------------------
  ----------------------------------------------------------
  -- return cursor/result set of all applications
  FUNCTION get_application_list RETURN security.return_refcursor;

  ----------------------------------------------------------
  ----------------------------------------------------------
  -- allocate a user to a role
  -- returns null if successful, or 'Error.....' if not
  FUNCTION insert_user_role(p_user_name IN CHAR,
                            p_user_role IN CHAR) RETURN CHAR;

  ----------------------------------------------------------
  ----------------------------------------------------------
  /* delete either:
            a single role from a user's role allocation
            all roles for a specific application
            all roles for user
  */
  FUNCTION delete_user_role(p_user_name       IN CHAR,
                            p_user_role       IN CHAR,
                            p_application_key IN CHAR) RETURN CHAR;

  ----------------------------------------------------------
  ----------------------------------------------------------
  /* return all users that are assigned p_role */
  FUNCTION get_all_users_by_role(p_role IN CHAR) RETURN return_refcursor;

  ----------------------------------------------------------
  ----------------------------------------------------------
  /* return all locations that are assigned to a certain (resort)user */
  FUNCTION get_all_locations_for_user(p_user_name IN CHAR)
    RETURN return_refcursor;

  ----------------------------------------------------------
  ----------------------------------------------------------
  /* delete all location assignments for a given (resort)user */
  FUNCTION delete_user_location(p_user_name IN CHAR) RETURN CHAR;

  ----------------------------------------------------------
  ----------------------------------------------------------
  /* insert location assignment for a given (resort)user */
  FUNCTION insert_user_location(p_user_name     IN CHAR,
                                p_user_location IN CHAR) RETURN CHAR;

END security;
