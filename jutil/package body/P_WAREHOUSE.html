PACKAGE BODY p_Warehouse IS 

  -- ************************************************************
  -- * Function return exchange rate of a selected currency     *
  -- ************************************************************

  FUNCTION Get_Uk_Exchange_Rate (p_Currency    IN currency_rate.currency_code%TYPE,
                                 p_SeasonType  IN CHAR,
                                 p_SeasonYear  IN CHAR)
  
  RETURN currency_rate.uk_exchange_rate%TYPE IS
  
    v_rate       currency_rate.uk_exchange_rate%TYPE;
  BEGIN
  
    BEGIN
      SELECT cr.uk_exchange_rate
      INTO   v_rate
      FROM   currency_rate cr
      WHERE  cr.currency_code = p_Currency
      AND    cr.season_year   = p_SeasonYear
      AND    cr.season_type   = p_SeasonType;
      
      EXCEPTION
      WHEN no_data_found THEN
        v_rate := 1;
    END;  
    
    RETURN(v_rate);
  END;

BEGIN 
  -- Initialization
  NULL;
END p_Warehouse;
