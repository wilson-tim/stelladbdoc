PACKAGE BODY p_util_maint IS

  -- Author  : Lars Svensson
  -- Created : 21/05/03
  -- Purpose : Defines the generic stored procedures / functions that are used
  --         : by java programs to read data from Oracle



----------------------------------------------------------------
----------------------------------------------------------------                      
-- Gets a single row of data from the MAINT_GUI_PARAMETERS table
-- and returns it as result set
FUNCTION get_maint_gui_parameters( p_appl_key CHAR, p_table_name CHAR )
                      RETURN p_util_maint.return_refcursor

IS

  v_result p_util_maint.return_refcursor;
  
BEGIN

  OPEN v_result FOR
       SELECT *
       FROM  maint_gui_parameters
       WHERE application_key = p_appl_key AND table_name = p_table_name;
      
  RETURN(v_result);
END; -- get_maint_gui_parameters



END p_util_maint; -- end of package body
