FUNCTION get_seasons RETURN security.return_refcursor IS

  -- Author  : Leigh Ashton
  -- Created : 08/7/04
  -- Purpose : return cursor of all season info from season table
  --         :
  --         :

  v_result security.return_refcursor;

BEGIN
  OPEN v_result FOR
  
    SELECT season_year,
           season_type,
           CASE
             WHEN SYSDATE BETWEEN season_start_date AND season_end_date THEN
              1
             ELSE
              0
           END current_season,
           season_start_date,
           season_end_date,
           season_type || substr(season_year, 3, 4) AS season_code
      FROM season
     ORDER BY season_year, season_type;

  RETURN(v_result);

END;
