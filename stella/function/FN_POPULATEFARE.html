FUNCTION fn_populatefare(recIdToProcess  IN VARCHAR2, 
                                           oneLine         IN VARCHAR2) RETURN VARCHAR2 AS

-- Read and process each individual file 

        startPos  NUMBER:= 0;
        pos1      NUMBER:= 0;
        pos2      NUMBER:= 0;
        fareAmt   NUMBER :=0;
        fareFound BOOLEAN:= false;
BEGIN

-- K-F and K-I,K-B,KN-F and KN-I,KN-B
    startPos := LENGTH(recIdToProcess);

-- get the position of first and second delimiter (;) if it is
-- equivalent currency case
    pos1 := INSTR(oneLine,';',1,1);
    pos2 := INSTR(oneLine,';',1,2);

    IF SUBSTR(oneLine,startPos+1,3) = 'GBP' THEN
       fareAmt   := TO_NUMBER(SUBSTR(oneLine,startPos + 4, pos1 - (startPos+4)));
       fareFound := true;
    END IF;
  
    IF NOT fareFound THEN
-- In case of Equivalent currency
       IF SUBSTR(oneLine,pos1+1, 3) = 'GBP' THEN
          fareAmt := TO_NUMBER(SUBSTR(oneLine,pos1 + 4, pos2-pos1));
       END IF;
    END IF;				

    RETURN(fareAmt);
EXCEPTION
WHEN OTHERS THEN 
    RETURN(0);
END fn_populatefare;
